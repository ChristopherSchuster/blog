Scala es un lenguage de programación objecto-funcional, los programas escritos en este lenguaje se ejecutan en la máquina virtual de Java lo que hace posible desarrollar programas híbridos usando componentes de Java y Scala, este post es una breve introducción.

Instalación 

    sudo apt-get install scala

Hola mundo

Crearemos una archivo llamado HelloWorld.scala con el siguiente código: 

    object HelloWorld {
      def main(args: Array[String]) {
        println("Hello, world!")
      }
    }

Podemos compilar el programa con el comando scalac HelloWorld.scala , este comando generará una clase llamada HelloWorld.class, podemos ejecutar el programa con el comando scala HelloWorld

Simple Servidor TCP echo usando java.net.*

El código a continuación implmenta un servidor tcp en java: 


    import java.lang.*;
    import java.io.*;
    import java.net.*;

    class Server {
        public static void main(String args[]) {
            try {
                int serverPort = 4020;
                ServerSocket serverSocket = new ServerSocket(serverPort);
                serverSocket.setSoTimeout(10000);

                while(true) {
                    System.out.println("Waiting for client on port " + serverSocket.getLocalPort() + "...");

                    Socket server = serverSocket.accept();
                    System.out.println("Just connected to " + server.getRemoteSocketAddress());

                    PrintWriter toClient =
                        new PrintWriter(server.getOutputStream(),true);
                    BufferedReader fromClient =
                        new BufferedReader(
                                new InputStreamReader(server.getInputStream()));
                    String line = fromClient.readLine();
                    System.out.println("Server received: " + line);
                    toClient.println(line);
                    server.close();
                }

            } catch(Exception e) {
                System.out.print(e.getMessage());
            }
        }
    }

Podemos escribir este mismo programa en scala usando el paquete java.net.* de java: 


import java.net.{ ServerSocket, SocketException, SocketTimeoutException, Socket }
import java.io.{ PrintWriter, BufferedReader, InputStreamReader, OutputStreamWriter }

object EchoServer {
  var port = 1320

  def run() {
    val serverSocket = new ServerSocket(port)

    def process( socket: Socket) {
      val bufferedReader = new BufferedReader(new InputStreamReader(socket.getInputStream))
      var line = ""
      var msg = ""
        do {
          line = bufferedReader.readLine()
          msg += line + "\n"
        } while (line != "")

      val out: PrintWriter = new PrintWriter(new OutputStreamWriter(socket.getOutputStream))
      out.println(msg)
      out.flush()
    }

    def loop() {
      while (true) {
        val socket = serverSocket.accept()
        process(socket)
        socket.close()
      }
    }
    loop()
  }

  def main(args: Array[String]) {

    if(args.length > 0) {
      val arglist = args.toList
      port = arglist(0).toInt
    }
    run()
  }

}
